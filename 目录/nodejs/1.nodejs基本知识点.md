# I.nodejs基本知识点：

1.nodejs常用API

A.http:服务器请求

```js
//1.导入http模块
const http = require('http');
//2.创建服务器
const server = http.createServer((req, res) =>{
    res.end('hello world');
})
//3.开启服务器
server.listen(3000, ()=>{
    console.log('success');
});
```

B.path：文件路径

```js
const path = require('path');
__dirname: 文件路径 = 文件全路径-当前自身的路径
__filename: 文件全路径
// path的文件路径拼接
path.join(__dirname, 'zero','1.txt');
```

C.fs：文件读写

```js
const fs = require('fs');
// 读取文件
fs.readFile
// 写入文件
fs.writeFile
```



2.nodejs配置自己的服务器：

```js
const http = require('http');

const path = require('path');

const fs = require('fs');

const server = http.createServer((req, res) =>{
    // 获取到文件地址     req.url：获取到用户请求的文件名
    const fullFile = path.join(__dirname, 'zero', req.url);
    // 将文件地址
    // fs.readFile(fullFile, 'utf-8', (err, data)=>{
    fs.readFile(fullFile, (err, data)=>{
        // 发送数据
        // 判断数据发送是否成功
        if(err == null){
            res.end(data);
        }else{
            // 设置编码格式转义
            res.setHeader('Content-type','text/html;charset=utf8');
            res.end('404  页面不存在');
        }
    });
});

server.listen(3035, ()=>{
    console.log('success zero.js');
});
```
